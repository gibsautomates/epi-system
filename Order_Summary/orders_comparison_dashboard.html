üìã SQL Query 1: Daily Summary Comparison
-- Copy this query and run it in Supabase SQL Editor WITH daily_summary AS ( SELECT export_date, COUNT(DISTINCT "OrderID") as total_orders, SUM("Qty") as total_units, SUM("GrandTotal") as total_revenue, AVG("GrandTotal") as avg_order_value, COUNT(DISTINCT "ProductID") as unique_products, COUNT(DISTINCT CONCAT("CompanyID", "OrderSource")) as active_channels FROM orders_daily WHERE export_date IN ( '2025-09-10', -- Replace with your actual dates '2025-09-09', -- Yesterday '2025-09-03' -- Same day last week ) GROUP BY export_date ) SELECT CASE WHEN export_date = '2025-09-10' THEN 'Today' WHEN export_date = '2025-09-09' THEN 'Yesterday' WHEN export_date = '2025-09-03' THEN 'Last Week' END as period, export_date, total_orders, total_units, ROUND(total_revenue, 2) as total_revenue, ROUND(avg_order_value, 2) as avg_order_value, unique_products, active_channels FROM daily_summary ORDER BY export_date DESC;
üìã SQL Query 2: Channel Comparison
-- Copy this query and run it in Supabase SQL Editor WITH channel_comparison AS ( SELECT export_date, CONCAT("CompanyID", "OrderSource") as channel_code, COALESCE(cl.channel_name, 'Unknown') as channel_name, COUNT(DISTINCT "OrderID") as orders, SUM("Qty") as units, ROUND(SUM("GrandTotal"), 2) as revenue FROM orders_daily od LEFT JOIN channel_lookup cl ON CONCAT(od."CompanyID", od."OrderSource") = cl.channel_code WHERE export_date IN ('2025-09-10', '2025-09-09', '2025-09-03') -- Update dates GROUP BY export_date, channel_code, channel_name ) SELECT channel_code, channel_name, COALESCE(MAX(CASE WHEN export_date = '2025-09-10' THEN orders END), 0) as today_orders, COALESCE(MAX(CASE WHEN export_date = '2025-09-10' THEN revenue END), 0) as today_revenue, COALESCE(MAX(CASE WHEN export_date = '2025-09-09' THEN revenue END), 0) as yesterday_revenue, COALESCE(MAX(CASE WHEN export_date = '2025-09-03' THEN revenue END), 0) as lastweek_revenue, ROUND( CASE WHEN MAX(CASE WHEN export_date = '2025-09-09' THEN revenue END) > 0 THEN ((MAX(CASE WHEN export_date = '2025-09-10' THEN revenue END) - MAX(CASE WHEN export_date = '2025-09-09' THEN revenue END)) / MAX(CASE WHEN export_date = '2025-09-09' THEN revenue END) * 100) ELSE 0 END, 1) as revenue_change_vs_yesterday, ROUND( CASE WHEN MAX(CASE WHEN export_date = '2025-09-03' THEN revenue END) > 0 THEN ((MAX(CASE WHEN export_date = '2025-09-10' THEN revenue END) - MAX(CASE WHEN export_date = '2025-09-03' THEN revenue END)) / MAX(CASE WHEN export_date = '2025-09-03' THEN revenue END) * 100) ELSE 0 END, 1) as revenue_change_vs_lastweek FROM channel_comparison GROUP BY channel_code, channel_name ORDER BY today_revenue DESC;
üìã SQL Query 3: Top Products
-- Copy this query and run it in Supabase SQL Editor WITH product_comparison AS ( SELECT export_date, "ProductID", "DisplayName", SUM("Qty") as units_sold, ROUND(SUM("LineTotal"), 2) as revenue FROM orders_daily WHERE export_date IN ('2025-09-10', '2025-09-09', '2025-09-03') -- Update dates GROUP BY export_date, "ProductID", "DisplayName" ) SELECT "ProductID", LEFT("DisplayName", 50) as product_name, COALESCE(MAX(CASE WHEN export_date = '2025-09-10' THEN units_sold END), 0) as today_units, COALESCE(MAX(CASE WHEN export_date = '2025-09-10' THEN revenue END), 0) as today_revenue, COALESCE(MAX(CASE WHEN export_date = '2025-09-09' THEN revenue END), 0) as yesterday_revenue, COALESCE(MAX(CASE WHEN export_date = '2025-09-03' THEN revenue END), 0) as lastweek_revenue FROM product_comparison GROUP BY "ProductID", "DisplayName" HAVING (today_units + COALESCE(MAX(CASE WHEN export_date = '2025-09-09' THEN units_sold END), 0)) > 0 ORDER BY today_revenue DESC LIMIT 15;
